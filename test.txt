//print all numbers from 1 - 100,
//multiple 3 fizz, multiple 5 buzz, 3&5 fizzbuzz

//take our range of numbers, printing all of them
//    some sort of loop
// if multiple % === 0 fizz
//  else if multiple 5 % buzz, 
// 3&5 both

const fizzBuzzer = function (num, num2) {

  for (let i = num; i <= num2; i++) {
    if (i % 3 === 0 && i % 5 === 0) {
      console.log('fizzbuzz');
    } else if (i % 3 === 0) {
      console.log(fizz);
    } else if (i % 5 === 0) {
      console.log(buzz);
    } else {
      console.log(num);
    }
  }

};

fizzBuzzer(1, 100);
